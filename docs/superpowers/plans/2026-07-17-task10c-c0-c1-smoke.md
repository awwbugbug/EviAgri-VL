# Task 10C C0+C1 Diagnosis-Only Smoke Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建并执行Task 10C训练前协议门控和三种子8-step diagnosis-only QLoRA工程smoke，在完整报告后强制停止于64-step C2之前。

**Architecture:** 复用Task 10B v2的已签名manifest，但生成独立的Task 10C对话记录与smoke子集；训练、推理、评估分别由独立模块负责，并以SHA256和全新输出目录串联。C1只验证训练/保存/重载/推理/严格解析闭环，性能指标只观察不调参，最终决策永远不授权自动执行C2。

**Tech Stack:** Python 3.10、PyTorch 2.5.1、Transformers 4.51.3、PEFT 0.15.2、bitsandbytes 0.45.5、Qwen2.5-VL-3B-Instruct、pytest、Bash、SHA256。

## Global Constraints

- 只实现和执行C0+C1；禁止实现或启动64-step C2、Task10D、官方test、Task8 locked内容、AGE、7B、新backbone、动态LoRA/Gating或SAM2。
- 输入manifest SHA256必须为`84d2d1b20d4a781bc6fca8c4e9c41dd57051b6201287193681409451095edc90`。
- exact split固定16类、train/val/dev=`192/48/80`、每类=`12/3/5`；不得重新抽样主split。
- smoke train为每类按source SHA升序前4张，共64张；smoke dev为每类按source SHA升序第1张，共16张。
- seeds固定`17,29,43`；每seed从同一Base独立训练8 optimizer steps、64 exposures。
- QLoRA固定NF4、`r=16`、`alpha=32`、`dropout=0.05`、targets=`q_proj,v_proj`。
- batch=`1`、gradient accumulation=`8`、lr=`1e-4`、loss reduction=`per_example_active_token_mean_then_batch_mean`。
- processor固定`min_pixels=200704`、`max_pixels=401408`；解码固定greedy、`max_new_tokens=32`。
- smoke不得选择seed、修改超参数、设置性能门槛或续跑为正式模型。
- 每个写入器拒绝已有非空目录；失败保留`failure.json`；完整成功才写`completion.sha256`。
- 服务器保持开机；任何异常先保留现场并只读诊断。

```python
CLASS_IDS = (9, 10, 16, 17, 22, 24, 45, 50, 64, 68, 71, 82, 83, 87, 99, 101)

def canonical_pest_id(class_id: int) -> str:
    if class_id not in CLASS_IDS:
        raise ValueError(f"class is outside frozen Task 10C set: {class_id}")
    return f"IP{class_id:03d}"


def strict_parse_pest_json(raw: str) -> dict:
    try:
        value = json.loads(raw)
    except json.JSONDecodeError:
        return {"syntax_valid": False, "schema_valid": False, "pest_id": None, "error": "invalid_json"}
    valid = (
        isinstance(value, dict)
        and list(value) == ["pest_id"]
        and isinstance(value["pest_id"], str)
        and value["pest_id"] in {f"IP{x:03d}" for x in CLASS_IDS}
        and raw == json.dumps(value, separators=(",", ":"))
    )
    return {
        "syntax_valid": True,
        "schema_valid": valid,
        "pest_id": value.get("pest_id") if valid else None,
        "error": None if valid else "schema_or_canonical_format",
    }
```

```python
def validate_diagnosis_record(record: dict) -> None:
    messages = record["messages"]
    if [m.get("role") for m in messages] != ["system", "user", "assistant"]:
        raise ValueError("Task 10C requires exact system/user/assistant messages")
    if messages[0]["content"][0]["text"] != SYSTEM_PROMPT:
        raise ValueError("Task 10C system prompt mismatch")
    if [x.get("type") for x in messages[1]["content"]] != ["image", "text"]:
        raise ValueError("Task 10C user content must be image then text")
    if messages[1]["content"][1]["text"] != TRAIN_PROMPT:
        raise ValueError("Task 10C training prompt mismatch")
    parsed = strict_parse_pest_json(messages[2]["content"][0]["text"])
    if not parsed["schema_valid"]:
        raise ValueError("Task 10C assistant target is invalid")


def compute_task10c_loss(model, inputs):
    values = dict(inputs)
    labels = values.pop("labels")
    outputs = model(**values)
    logits = outputs["logits"] if isinstance(outputs, dict) else outputs.logits
    loss, audit = per_example_active_token_mean(logits, labels)
    return loss, outputs, audit


class Task10CTrainer(Trainer):
    def compute_loss(self, model, inputs, return_outputs=False, num_items_in_batch=None):
        del num_items_in_batch
        loss, outputs, _ = compute_task10c_loss(model, inputs)
        return (loss, outputs) if return_outputs else loss
```

```python
def run_directory(root: str | Path, seed: int) -> Path:
    if seed not in {17, 29, 43}:
        raise ValueError("Task 10C smoke requires frozen seed")
    return Path(root) / "training" / f"seed_{seed}"


def verify_protocol_gate(protocol_root: str | Path) -> dict:
    root = Path(protocol_root)
    verify_completion_file(root / "completion.sha256", root)
    report = json.loads((root / "preflight_report.json").read_text(encoding="utf-8"))
    if report.get("passed") is not True or report.get("manifest_sha256") != EXPECTED_MANIFEST_SHA256:
        raise ValueError("Task 10C protocol gate is not passed")

    return report
```

```python
def decide_c1_engineering(seed_reports: Mapping[int, dict]) -> dict:
    reasons = []
    if set(seed_reports) != {17, 29, 43}:
        reasons.append("missing frozen seed")
    for seed, report in sorted(seed_reports.items()):
```
